@create Parent: Stripper=10
@Desc Parent: Stripper=%R%R%T[u(u(sex)-desc)]%R%R[center(Type 'Stripper-help' for help,79)]%R[switch(owner(me),%#,[center(Type 'Owner-help' for configuration help,79)])]
&MALE-DESC Parent: Stripper=This is the male description for the stripper, type &MALE-DESC [num(me)]=<description> to set my description.
&FEMALE-DESC Parent: Stripper=This is the female description for the stripper, type &FEMALE-DESC [num(me)]=<description> to set my description.
@Sex Parent: Stripper=Male
&CMD_STRIPPER_HELP Parent: Stripper=$STRIPPER-HELP:@pemit %#=%ch%crStripper Code %cw[u(Version)] %crby [u(Credits)]%cn%R%R[ljust(%chStripper-help%cn,20)]- This help menu.%R[ljust(%chtip <pose>%cn,20)]- Tip the stripper, with <pose>.%R[ljust(%chflirt <pose>%ch,20)]- Flirt with stripper, by using <pose>.%R%RExample of useage:%R%chtip%cn tucks a few bills under [name(me)]'s g-string and pats [switch(u(sex),MALE,his,FEMALE,her)] behind.%r%RThe stripper will pose.%R[name(%#)] tucks a few bills under [name(me)]'s g-string and pats [switch(u(sex),MALE,his,FEMALE,her)] behind.%R%RA few seconds after the initial command usage, the stripper will pose a reaction.%R
&VERSION Parent: Stripper=V2.0
&CREDITS Parent: Stripper=Cole@MedNights
&CMD_FLIRT Parent: Stripper=$flirt *:@emit [name(%#)] %0;@wait 5=@emit [u(FLIRT_[u(SEX)]_[u(%#/SEX)]_[u(Orientation)]_[inc(rand(words(lattr(me/FLIRT_[u(sex)]*))))])]
&CMD_TIP Parent: Stripper=$tip *:@emit [name(%#)] %0;@wait 5=@emit [u(TIP_[u(SEX)]_[u(%#/SEX)]_[u(Orientation)]_[inc(rand(words(lattr(me/TIP_[u(sex)]*))))])]
@Listen Parent: Stripper=*
@Ahear Parent: Stripper=@switch [match(%#,#6)][match([last(%0,%b)],arrived.)]=01,{@emit [u(POSE_[u(SEX)]_[inc(rand(words(lattr(me/POSE_[u(sex)]*))))])]},00,{@switch [gte([match(%0,*[name(me)]*)],1)]=1,{@emit [u(HEARD_NAME_[u(SEX)]_[u(%#/SEX)]_[inc(rand(words(lattr(me/HEARD_NAME_[u(sex)]_[u(%#/SEX)]*))))])]}}
&HEARD_NAME_MALE_MALE_1 Parent: Stripper=Random Response Male to Male Heard Name Used 1
&HEARD_NAME_FEMALE_MALE_1 Parent: Stripper=Random Response Female to Male Heard Name Used 1
&HEARD_NAME_MALE_FEMALE_1 Parent: Stripper=Random Response Male to Female Heard Name Used 1
&HEARD_NAME_FEMALE_FEMALE_1 Parent: Stripper=Random Response Female to Female Heard Name Used 1
&POSE_MALE_1 Parent: Stripper=Random Arrival Pose for Male Stripper 1
&POSE_FEMALE_1 Parent: Stripper=Random Arrival Pose for Female Stripper 1
&CMD_OWNER_HELP Parent: Stripper=$OWNER-HELP:@switch [owner(me)]=%#,{@pemit %#=%chOwner Help for Stripper %cn[u(version)]%ch by [u(Credits)]%cn%R%RThe following help topics are available for helping you create a customized stripper.%R%R%ch1%cn - Flirting Reactions.%R%ch2%cn - Tip reactions.%R%ch3%cn - Responses to hearing the strippers name.%R%ch4%cn - Arrival Poses.%R%ch5%cn - Gender Determinations.%R%ch6%cn - Descriptions.%R%ch7%cn - Setting attributes.%R%ch8%cn - Substitutions.%R%ch9%cn - Orientation options.%R%RType %chOwner-Help #%cn for details.}
&CMD_OWNER_HELP_OPTION Parent: Stripper=$OWNER-HELP *:@switch [owner(me)]=%#,{@pemit %#=%chOwner Help for Stripper%cn [u(Version)] %chby [u(Credits)]%cn%R%R[u(HELP_%0)]%R}
